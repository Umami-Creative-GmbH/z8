# Connected Masked Time Input Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Convert the shared webapp `TimeInput` into a connected masked text field with a picker trigger button, while preserving normalized `HH:mm` form values.

**Architecture:** Keep `TimeInput` as the single integration point. Add `react-imask` for masked editing, keep `timepicker-ui` for the popup picker, and attach the picker to a hidden internal input so the visible text input never opens the picker directly. Maintain a small set of parse/format helpers inside `time-input.tsx` because the behavior is specific to this component.

**Tech Stack:** Next.js 16, React 19, TypeScript, Vitest, Testing Library, `react-imask`, `timepicker-ui`, Tailwind CSS, `@tabler/icons-react`, pnpm.

- [ ] **Step 2: Add local helper functions**

Add these helpers below `normalizePickerConfirmTime`:

```tsx
function createChangeEvent(value: string): React.ChangeEvent<HTMLInputElement> {
	const changeTarget = { value } as HTMLInputElement;
	return {
		currentTarget: changeTarget,
		target: changeTarget,
		type: "change",
	} as React.ChangeEvent<HTMLInputElement>;
}

function getPeriodFromTime(value: string | number | readonly string[] | undefined): "AM" | "PM" {
	if (typeof value !== "string") {
		return "AM";
	}

	const hour = Number(value.split(":")[0]);
	return Number.isInteger(hour) && hour >= 12 ? "PM" : "AM";
}

function formatTimeForMaskedInput(value: string | number | readonly string[] | undefined, timeFormat: TimeFormat): string {
	if (typeof value !== "string" || !/^\d{2}:\d{2}$/.test(value)) {
		return typeof value === "string" ? value : "";
	}

	if (timeFormat === "24h") {
		return value;
	}

	const [hourPart, minutePart] = value.split(":");
	const hour = Number(hourPart);
	const displayHour = hour % 12 === 0 ? 12 : hour % 12;
	return `${displayHour.toString().padStart(2, "0")}:${minutePart}`;
}

function parseMaskedTime(value: string, timeFormat: TimeFormat, period: "AM" | "PM"): string | null {
	if (!/^\d{2}:\d{2}$/.test(value)) {
		return null;
	}

	const [hourPart, minutePart] = value.split(":");
	const hour = Number(hourPart);
	const minutes = Number(minutePart);

	if (!Number.isInteger(hour) || !Number.isInteger(minutes) || minutes < 0 || minutes > 59) {
		return null;
	}

	if (timeFormat === "24h") {
		if (hour < 0 || hour > 23) {
			return null;
		}

		return `${hour.toString().padStart(2, "0")}:${minutePart}`;
	}

	if (hour < 1 || hour > 12) {
		return null;
	}

	const storedHour = period === "PM" ? (hour === 12 ? 12 : hour + 12) : hour === 12 ? 0 : hour;
	return `${storedHour.toString().padStart(2, "0")}:${minutePart}`;
}
```

- [ ] **Step 3: Replace `TimeInput` internals**

Replace the current `TimeInput` function body with this implementation:

```tsx
function TimeInput({
	className,
	onBlur,
	onChange,
	value,
	defaultValue,
	timeFormat,
	disabled,
	...props
}: TimeInputProps) {
	const inputRef = useRef<HTMLInputElement>(null);
	const pickerAnchorRef = useRef<HTMLInputElement>(null);
	const onChangeRef = useRef(onChange);
	const isEditingRef = useRef(false);
	const reactId = useId().replace(/[^a-zA-Z0-9_-]/g, "");
	const modalRootId = `time-input-${reactId}`;
	const contextTimeFormat = useTimeFormat();
	const pickerFormat = normalizeTimeFormat(timeFormat ?? contextTimeFormat);
	const [period, setPeriod] = useState<"AM" | "PM">(() => getPeriodFromTime(value ?? defaultValue));
	const [draftValue, setDraftValue] = useState(() =>
		formatTimeForMaskedInput(value ?? defaultValue, pickerFormat),
	);

	onChangeRef.current = onChange;

	useEffect(() => {
		if (isEditingRef.current) {
			return;
		}

		setDraftValue(formatTimeForMaskedInput(value ?? defaultValue, pickerFormat));
		setPeriod(getPeriodFromTime(value ?? defaultValue));
	}, [defaultValue, pickerFormat, value]);

	useEffect(() => {
		if (!pickerAnchorRef.current) {
			return;
		}

		const picker = new TimepickerUI(pickerAnchorRef.current, {
			clock: {
				type: timeFormatToPickerType(pickerFormat),
			},
			ui: {
				appendModalSelector: `#${modalRootId}`,
				editable: false,
			},
			callbacks: {
				onConfirm: (data) => {
					const nextValue = normalizePickerConfirmTime(data, pickerFormat);
					if (!nextValue) {
						return;
					}

					isEditingRef.current = false;
					setDraftValue(formatTimeForMaskedInput(nextValue, pickerFormat));
					setPeriod(getPeriodFromTime(nextValue));
					onChangeRef.current?.(createChangeEvent(nextValue));
				},
			},
		});

		picker.create();

		return () => picker.destroy({ keepInputValue: true });
	}, [modalRootId, pickerFormat]);

	const emitIfValid = (nextDisplayValue: string, nextPeriod = period) => {
		const nextValue = parseMaskedTime(nextDisplayValue, pickerFormat, nextPeriod);
		if (!nextValue) {
			return;
		}

		onChangeRef.current?.(createChangeEvent(nextValue));
	};

	const handleAccept = (nextValue: unknown) => {
		const nextDisplayValue = typeof nextValue === "string" ? nextValue : "";
		isEditingRef.current = true;
		setDraftValue(nextDisplayValue);
		emitIfValid(nextDisplayValue);
	};

	const handleBlur = (event: React.FocusEvent<HTMLInputElement>) => {
		isEditingRef.current = false;
		setDraftValue(formatTimeForMaskedInput(value ?? defaultValue, pickerFormat));
		onBlur?.(event);
	};

	const handlePeriodToggle = () => {
		const nextPeriod = period === "AM" ? "PM" : "AM";
		setPeriod(nextPeriod);
		emitIfValid(draftValue, nextPeriod);
	};

	const handlePickerButtonClick = () => {
		pickerAnchorRef.current?.click();
	};

	const maskBlocks =
		pickerFormat === "24h"
			? {
					HH: { mask: IMask.MaskedRange, from: 0, to: 23 },
					mm: { mask: IMask.MaskedRange, from: 0, to: 59 },
				}
			: {
					hh: { mask: IMask.MaskedRange, from: 1, to: 12 },
					mm: { mask: IMask.MaskedRange, from: 0, to: 59 },
				};

	return (
		<div className="contents" id={modalRootId}>
			<div
				className={cn(
					"flex h-9 w-full min-w-0 overflow-hidden rounded-md border border-input bg-transparent shadow-xs transition-[color,box-shadow] focus-within:border-ring focus-within:ring-[3px] focus-within:ring-ring/50 dark:bg-input/30",
					props["aria-invalid"] && "border-destructive ring-destructive/20 dark:ring-destructive/40",
					disabled && "pointer-events-none cursor-not-allowed opacity-50",
					className,
				)}
				data-slot="time-input"
			>
				<IMaskInput
					{...props}
					blocks={maskBlocks}
					className="min-w-0 flex-1 bg-transparent px-3 py-1 text-base outline-none selection:bg-primary selection:text-primary-foreground placeholder:text-muted-foreground md:text-sm"
					disabled={disabled}
					inputRef={inputRef}
					mask={pickerFormat === "24h" ? "HH:mm" : "hh:mm"}
					onAccept={handleAccept}
					onBlur={handleBlur}
					type="text"
					value={draftValue}
				/>
				{pickerFormat === "12h" ? (
					<button
						aria-label={`Switch to ${period === "AM" ? "PM" : "AM"}`}
						className="border-input border-l px-2 font-medium text-muted-foreground text-xs transition-colors hover:bg-accent hover:text-accent-foreground focus-visible:border-ring focus-visible:ring-[3px] focus-visible:ring-ring/50 disabled:pointer-events-none"
						disabled={disabled}
						onClick={handlePeriodToggle}
						type="button"
					>
						{period}
					</button>
				) : null}
				<Button
					aria-label="Open time picker"
					className="h-full rounded-none border-input border-l px-2.5 shadow-none"
					disabled={disabled}
					onClick={handlePickerButtonClick}
					size="icon"
					type="button"
					variant="ghost"
				>
					<IconClock className="size-4" aria-hidden="true" />
				</Button>
			</div>
			<input
				aria-hidden="true"
				className="sr-only"
				disabled={disabled}
				readOnly
				ref={pickerAnchorRef}
				tabIndex={-1}
				type="text"
				value={draftValue}
			/>
		</div>
	);
}
```
