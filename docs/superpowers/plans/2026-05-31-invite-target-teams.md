# Invite Target Teams Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add optional target team assignment to direct email invitations and align invite-code team assignment language and behavior.

**Architecture:** Store direct invite target teams on Better Auth invitations as `targetTeamId`, validate all team IDs server-side against `organizationId`, and pass valid target teams into employee provisioning. Keep invite-code persistence as `defaultTeamId`, but present it as “Target team” and ensure approval/no-approval flows apply the same intent.

**Tech Stack:** Next.js App Router, Better Auth organization plugin, Drizzle ORM/Postgres, Effect services, React, TanStack Query, TanStack Form, Tolgee, Vitest, pnpm.

```ts
export async function ensureEmployeeForOrganizationMember(
	dbClient: EmployeeProvisioningDb,
	input: {
		userId: string;
		organizationId: string;
		memberRole: OrganizationMemberRole;
		targetTeamId?: string | null;
	},
) {
	const existingEmployee = await dbClient.query.employee.findFirst({
		where: and(
			eq(employee.userId, input.userId),
			eq(employee.organizationId, input.organizationId),
		),
	});

	if (existingEmployee) {
		if (!existingEmployee.isActive && !existingEmployee.teamId && input.targetTeamId) {
			await dbClient
				.update(employee)
				.set({ isActive: true, teamId: input.targetTeamId })
				.where(eq(employee.id, existingEmployee.id));
		}

		return existingEmployee;
	}

	const insertResult = dbClient.insert(employee).values({
		userId: input.userId,
		organizationId: input.organizationId,
		teamId: input.targetTeamId ?? null,
		role: hasAdminOrganizationRole(input.memberRole) ? "admin" : "employee",
		isActive: true,
	});
```

Keep the existing admin `teamPermissions` block unchanged after the insert.

```ts
async function resolveValidTargetTeamId(
	dbService: typeof DatabaseService.Service,
	organizationId: string,
	targetTeamId: string | null | undefined,
) {
	if (!targetTeamId) {
		return null;
	}

	const targetTeam = await dbService.query("validateInvitationTargetTeam", async () => {
		return await db.query.team.findFirst({
			where: and(eq(team.id, targetTeamId), eq(team.organizationId, organizationId)),
		});

	if (!targetTeam) {
		throw new ValidationError({
			message: "Target team not found in this organization",
			field: "targetTeamId",
			value: targetTeamId,
		});
	}

	return targetTeam.id;
}
```

Add this export in `actions.ts` after `cancelInvitation`:

```ts
export async function updateInvitationTargetTeam(data: {
	invitationId: string;
	organizationId: string;
	targetTeamId?: string | null;
}): Promise<ServerActionResult<void>> {
	const effect = Effect.gen(function* (_) {
		const authService = yield* _(AuthService);
		const session = yield* _(authService.getSession());
		const dbService = yield* _(DatabaseService);

		const invitation = yield* _(
			dbService.query("getInvitationForTargetTeamUpdate", async () => {
				return await db.query.invitation.findFirst({
					where: and(
						eq(authSchema.invitation.id, data.invitationId),
						eq(authSchema.invitation.organizationId, data.organizationId),
						eq(authSchema.invitation.status, "pending"),
					),
				});
			}),
			Effect.flatMap((invitationRecord) =>
				invitationRecord
					? Effect.succeed(invitationRecord)
					: Effect.fail(
							new NotFoundError({
								message: "Pending invitation not found",
								entityType: "invitation",
								entityId: data.invitationId,
							}),
						),
			),
		);

		const memberRecord = yield* _(
			dbService.query("getCurrentMemberForInvitationTargetTeamUpdate", async () => {
				return await db.query.member.findFirst({
					where: and(
						eq(authSchema.member.userId, session.user.id),
						eq(authSchema.member.organizationId, invitation.organizationId),
					),
				});
			}),
		);

		if (!memberRecord || (memberRecord.role !== "admin" && memberRecord.role !== "owner")) {
			return yield* _(
				Effect.fail(
					new AuthorizationError({
						message: "Only admins and owners can update invitation target teams",
						userId: session.user.id,
						resource: "invitation",
						action: "update",
					}),
				),
			);
		}

		const targetTeamId = yield* _(
			Effect.tryPromise({
				try: () => resolveValidTargetTeamId(dbService, invitation.organizationId, data.targetTeamId),
				catch: (error) =>
					error instanceof ValidationError
						? error
						: new ValidationError({ message: "Invalid target team", field: "targetTeamId" }),
			}),
		);

		await db
			.update(authSchema.invitation)
			.set({ targetTeamId })
			.where(eq(authSchema.invitation.id, invitation.id));
	});

	return runServerActionSafe(effect.pipe(Effect.provide(AppLayer)));
}
```

```tsx
"use client";

import { IconLoader2 } from "@tabler/icons-react";
import { useMutation, useQuery, useQueryClient } from "@tanstack/react-query";
import { useTranslate } from "@tolgee/react";
import { useEffect, useState } from "react";
import { toast } from "sonner";
import { updateInvitationTargetTeam } from "@/app/[locale]/(app)/settings/organizations/actions";
import { listTeams } from "@/app/[locale]/(app)/settings/teams/actions";
import { ActionPanel, ActionPanelBody, ActionPanelContent, ActionPanelFooter, ActionPanelHeader, ActionPanelTitle } from "@/components/ui/action-panel";
import { Button } from "@/components/ui/button";
import { Label } from "@/components/ui/label";
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select";
import { queryKeys } from "@/lib/query";
import type { InvitationWithInviter } from "./organizations-page-client";

export function EditInvitationTargetTeamDialog({
	organizationId,
	invitation,
	open,
	onOpenChange,
}: {
	organizationId: string;
	invitation: InvitationWithInviter | null;
	open: boolean;
	onOpenChange: (open: boolean) => void;
}) {
	const { t } = useTranslate();
	const queryClient = useQueryClient();
	const [targetTeamId, setTargetTeamId] = useState(invitation?.targetTeamId ?? "none");
	useEffect(() => {
		if (open) {
			setTargetTeamId(invitation?.targetTeamId ?? "none");
		}
	}, [invitation?.id, invitation?.targetTeamId, open]);

	const { data: teamsResult } = useQuery({
		queryKey: queryKeys.teams.list(organizationId),
		queryFn: () => listTeams(organizationId),
		enabled: open,
	});
	const teams = teamsResult?.success ? teamsResult.data : [];

	const mutation = useMutation({
		mutationFn: async () => {
			if (!invitation) return;
			const result = await updateInvitationTargetTeam({
				organizationId,
				invitationId: invitation.id,
				targetTeamId: targetTeamId === "none" ? null : targetTeamId,
			});
			if (!result.success) throw new Error(result.error || "Failed to update invitation");
		},
		onSuccess: () => {
			queryClient.invalidateQueries({ queryKey: queryKeys.invitations.list(organizationId) });
			toast.success(t("organization.members.targetTeamUpdated", "Invitation target team updated"));
			onOpenChange(false);
		},
		onError: (error) => toast.error(error.message),
	});

	return (
		<ActionPanel open={open} onOpenChange={onOpenChange}>
			<ActionPanelContent>
				<ActionPanelHeader>
					<ActionPanelTitle>{t("organization.members.editTargetTeam", "Edit target team")}</ActionPanelTitle>
				</ActionPanelHeader>
				<ActionPanelBody className="space-y-2">
					<Label htmlFor="pendingInviteTargetTeam">{t("organization.invite.targetTeam", "Target team")}</Label>
					<Select value={targetTeamId} onValueChange={setTargetTeamId}>
						<SelectTrigger id="pendingInviteTargetTeam">
							<SelectValue />
						</SelectTrigger>
						<SelectContent>
							<SelectItem value="none">{t("organization.invite.noTargetTeam", "No team")}</SelectItem>
							{teams.map((team) => (
								<SelectItem key={team.id} value={team.id}>{team.name}</SelectItem>
							))}
						</SelectContent>
					</Select>
				</ActionPanelBody>
				<ActionPanelFooter>
					<Button type="button" variant="outline" onClick={() => onOpenChange(false)}>{t("common.cancel", "Cancel")}</Button>
					<Button type="button" disabled={mutation.isPending} onClick={() => mutation.mutate()}>
						{mutation.isPending && <IconLoader2 className="mr-2 size-4 animate-spin" />}
						{t("common.save", "Save")}
					</Button>
				</ActionPanelFooter>
			</ActionPanelContent>
		</ActionPanel>
	);
}
```
